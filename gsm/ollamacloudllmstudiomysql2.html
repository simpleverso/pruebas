MySQL Batch Processing

MySQL
📋 MySQL Setup Required:

Create a PHP proxy script on your server. Save the following as mysql_proxy.php:

View PHP Proxy Script 
<?php
header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Methods: POST, GET, OPTIONS');
header('Access-Control-Allow-Headers: Content-Type');

if ($_SERVER['REQUEST_METHOD'] === 'OPTIONS') exit(0);

$config = [
    'host' => $_POST['host'] ?? 'localhost',
    'username' => $_POST['username'] ?? 'root',
    'password' => $_POST['password'] ?? '',
    'database' => $_POST['database'] ?? '',
    'query' => $_POST['query'] ?? '',
    'action' => $_POST['action'] ?? 'select'
];

try {
    $pdo = new PDO(
        "mysql:host={$config['host']};dbname={$config['database']};charset=utf8mb4",
        $config['username'],
        $config['password'],
        [PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION]
    );

    if ($config['action'] === 'select') {
        $stmt = $pdo->prepare($config['query']);
        $stmt->execute();
        $results = $stmt->fetchAll(PDO::FETCH_ASSOC);
        echo json_encode(['success' => true, 'data' => $results, 'rowCount' => count($results)]);
    } 
    elseif ($config['action'] === 'update') {
        $updates = json_decode($_POST['updates'] ?? '[]', true);
        $successCount = 0;
        foreach ($updates as $update) {
            $stmt = $pdo->prepare($update['query']);
            $stmt->execute($update['params']);
            $successCount++;
        }
        echo json_encode(['success' => true, 'updated' => $successCount]);
    }
    elseif ($config['action'] === 'tables') {
        $stmt = $pdo->query("SHOW TABLES");
        $tables = $stmt->fetchAll(PDO::FETCH_COLUMN);
        echo json_encode(['success' => true, 'tables' => $tables]);
    }
    elseif ($config['action'] === 'describe') {
        $table = $_POST['table'] ?? '';
        $stmt = $pdo->query("DESCRIBE `$table`");
        $columns = $stmt->fetchAll(PDO::FETCH_ASSOC);
        echo json_encode(['success' => true, 'columns' => $columns]);
    }
} catch (Exception $e) {
    echo json_encode(['success' => false, 'error' => $e->getMessage()]);
}
?>
